korean_overflow_and_minor_check = {

	if = {
		limit = {
			check_variable = { temp_seats > 0 }
		}
		# loop until seats to be added is 0
		while_loop_effect = {
			limit = {
				check_variable = { temp_seats > 0 }
			}

			# loops through remaining parties, subtracting 1 seat each time
			for_loop_effect = {
				start = 0
				end = temp_array^size

				if = {
					limit = {
						check_variable = { KOR_parliament_seat_support_array^temp_array^v > 0 } # party must have more than 0 seats to be subtracted
						check_variable = { temp_seats > 0 } # seats to be added is bigger than 0 currently
					}
					subtract_from_variable = { KOR_parliament_seat_support_array^temp_array^v = 1 } # subtract from party 1 seat
					subtract_from_temp_variable = { temp_seats = 1 } # subtract from seats to be added
				}
			}
		}
	}
	# if its subtract
	else_if = {
		limit = {
			check_variable = { temp_seats < 0 }
		}
		# loop until seats to be subtracted is 0
		while_loop_effect = {
			limit = {
				check_variable = { temp_seats < 0 }
			}

			# loops through remaining parties, adding 1 seat each time
			for_loop_effect = {
				start = 0
				end = temp_array^size

				if = {
					limit = {
						check_variable = { KOR_parliament_seat_support_array^temp_array^v < KOR_parliament_main_array^size } # party must have less than max seats to be added
						check_variable = { temp_seats < 0 } # seats to be subtracted is less than 0 currently
					}
					add_to_variable = { KOR_parliament_seat_support_array^temp_array^v = 1 } # add to party 1 seat
					add_to_temp_variable = { temp_seats = 1 } # add to seats to be subtracted
				}
			}
		}
	}

}
