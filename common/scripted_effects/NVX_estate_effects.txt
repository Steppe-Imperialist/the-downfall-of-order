fourth_estate_islamist_increase = {
	custom_effect_tooltip = TT_fourth_estate_islamist_increase
	hidden_effect = {
		if = {
			limit = { has_idea = fourth_estate_islamist5 }
			add_popularity = {
				ideology = islamist
				popularity = 0.05
			}
		}
		if = {
			limit = { has_idea = fourth_estate_islamist4 }
			swap_ideas = {
				remove_idea = fourth_estate_islamist4
				add_idea = fourth_estate_islamist5
			}
		}
		if = {
			limit = { has_idea = fourth_estate_islamist3 }
			swap_ideas = {
				remove_idea = fourth_estate_islamist3
				add_idea = fourth_estate_islamist4
			}
		}
		if = {
			limit = { has_idea = fourth_estate_islamist2 }
			swap_ideas = {
				remove_idea = fourth_estate_islamist2
				add_idea = fourth_estate_islamist3
			}
		}
		if = {
			limit = { has_idea = fourth_estate_islamist1 }
			swap_ideas = {
				remove_idea = fourth_estate_islamist1
				add_idea = fourth_estate_islamist2
			}
		}
		if = {
			limit = { has_islamist_fourth_estate = no }

		}
	}
}
